### Spread Operator

```ts

/*
	Spread Operator
*/

const onlyString = ['1', '2', '3'];
const onlyNumbers = [1, 2, 3, 4];

const arr1 = [
	...onlyString,
] // string[] 로 유추됨

const arr2 = [
...onlyString,
...onlyNumbers
] // (number | string)[] 으로 유츄됨

```
