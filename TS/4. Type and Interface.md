
```ts

/*
	Type vs Interface

*/

interface IObject {
	x: number;
	y: number
}

type TObject = {
	x: number;
	y: number;
}


interface IFunction {
	(x:number, y:number): number
}

type TFunction (x:number, y:number) => number;


```
