


### 1) loop holes of any

```ts

/*
	Loopholes of Any
*/

let number: number;

number = 10;

number.toUpperCase(); // ERROR

(number as any).toUpperCase();

const multiplyTwo = (x: number, y: number) => {
	return x * y; 
}

let args1: any = 'ELMO KIM';
let args2: any = true;

multiplyTwo(args1, args2); // NO TS ERROR -> 런타임에 터짐

multiplyTwo(true, false) // TS ERROR

```
