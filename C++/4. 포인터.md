### 3. 포인터 practice

```cpp

int *p;
int i = 3, j;

p = &i;
j = *p;
j++;

printf("1. %d\n", *p); // 3
printf("2. %p\n", p); // i의 주소
printf("3. %d\n\n", *j); // 4

p = &j;

printf("4. %d\n", *p); // 4
printf("5. %p\n", p); // j주소

*p = *p + 3;

printf("6. %d\n", i); // 3
printf("7. %d\n", j); // 7
printf("8. %d\n", *p); // 7
```

### 4. Call By Value

```cpp

void toUpper(char c){
	if(c >= 'a' && c <= 'z'){
		c-= 32;
	}
}

int main(){
char ch;
printf("영문자 입력");
ch = getchar();
toUpper(ch);
printf("결과는 %c입니다.\n", ch);

return 0;
}

```

### 5. Call By Address 

```cpp

void toUpper(char* c){
	if(*c >= 'a' && *c <= 'z'){
		*c-= 32;
	}
}

int main(){
char ch;
printf("영문자 입력");
ch = getchar();
toUpper(&ch);
printf("결과는 %c입니다.\n", ch);

return 0;
}


```

전달하는 변수를 전달하려면 call by value
전달하는 변수를 변경하려면 call by address

```cpp

void swap(int *p1, int *p2){
	int temp;
	temp = *p1;
	*p1 = *p2;
	*p2 = temp;
}

int main(){
	int a=3, b=5;
	swap(&a, &b);
	printf("%d %d", a,b);
	
	return 0;
}

```
